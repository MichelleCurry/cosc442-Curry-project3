package game;

import static org.junit.jupiter.api.Assertions.*;

import java.awt.Color;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class InventoryTest {

	Inventory i;
	Item thing;
	
	@BeforeEach
	void setUp() throws Exception {
		//initialize the inventory and item
		i = new Inventory(3);
		thing = new Item ('*',Color.PINK , "crown","ugly");
	}

	@AfterEach
	void tearDown() throws Exception {
	}


	@Test
	void testAdd1() {
		//Add item and check if there with get()
		i.add(thing);
		Item exItem = thing;
		
		assertEquals(exItem, i.get(0));
	}
	
	@Test
	void testAdd2() {
		//Add item and check if there with get()
		i.add(thing);
		i.add(thing);
		Item exItem = thing;
		
		assertEquals(exItem, i.get(1));
	}
	
	@Test
	void testRemove() {
		//Add item then remove it to check if null
		i.add(thing);
		i.remove(thing);
		
		assertEquals(null, i.get(0));
	}
	
	void testRemoveNull() {
		//Add item then remove it to check if null
		i.remove(thing);
	}
	
	@Test
	void testisFull(){
		i.add(thing);
		
		assertEquals(false, i.isFull());

	}
	
	@Test
	void testcontainsTrue() {
		i.add(thing);
		
		assertEquals(true, i.contains(thing));
	}
	
	@Test
	void testcontainsFalse() {		
		assertEquals(false, i.contains(thing));
	}

}
